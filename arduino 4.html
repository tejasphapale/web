How It Works

This project aims to create a lamp that can be operated by a touch sensor. It incorporates cutting-edge touch technology to provide a convenient and modern lighting solution. The lamp's design will focus on both functionality and aesthetics, offering users a stylish and user-friendly lighting experience.

const int touchPin = 2;     // TTP223 OUT pin
const int lampPin = 8;      // LED or Relay pin

bool lampState = false;     // Initial state OFF
bool lastTouchState = LOW;  // To detect state change

void setup() {
    pinMode(touchPin, INPUT);
    pinMode(lampPin, OUTPUT);
    digitalWrite(lampPin, LOW); // Lamp OFF at start
    Serial.begin(9600);
}

void loop() {
    bool touchState = digitalRead(touchPin);
    
    // Detect rising edge (LOW to HIGH)
    if (touchState == HIGH && lastTouchState == LOW) {
        lampState = !lampState; // Toggle state
        digitalWrite(lampPin, lampState ? HIGH : LOW);
        Serial.println(lampState ? "Lamp ON" : "Lamp OFF");
        delay(300); // Debounce delay
    }
    
    lastTouchState = touchState;
}
